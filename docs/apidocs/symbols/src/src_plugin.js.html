 10  * Fires a contentChange event when editor content changes.

 12 (function() {

 14   /**
 15    * Base class for change obserers.
 16    * 
 17    * @class
 18    */
 19   ChangeObserver = CKEDITOR.tools.createClass({
 20     /**
 21      * Creates a ChangeObserver class instance.
 22      *
 23      * @constructor
 24      * @param {CKEDITOR.editor} editor object observer observes.
 25      */
 26     $ : function(editor) {
 27       this._editor = editor;
 28       this._oldContent = '';
 29       this._paused = false;
 30     },
 31     proto : /** @lends ChangeObserver.prototype */ {
 32       /**
 33        * Returns editor observer observes
 34        * 
 35        * @returns {CKEDITOR.editor} editor that observer observes
 36        */
 37       getEditor : function() {
 38         return this._editor;
 39       },
 40       /**
 41        * Checks whether editor content has changed and fires a contentChange event if it has.
 42        * 
 43        * Event contains oldContent and currentContent properties that contain content before and after the change.
 44        */
 45       checkChange : function() {
 46         if ((this._paused == false) && (this.getEditor().readOnly == false)) {
 47 
 48           var currentContent = this._getEditorContent();
 49           if (this._oldContent != currentContent) {
 50             this._editor.fire("contentChange", {
 51               oldContent : this._oldContent,
 52               currentContent : currentContent
 53             });
 54 
 55             this._oldContent = currentContent;
 56           }
 57         }
 58       },
 59       /**
 60        * Stops observer
 61        */
 62       disconnect : function() {
 63       },
 64       /**
 65        * Starts observer
 66        */
 67       start : function() {
 68       },
 69       /**
 70        * Resets observer state. This means that observer assumes that content has not changed.
 71        */
 72       reset : function() {
 73         this._oldContent = this._getEditorContent();
 74       },
 75       /**
 76        * Pauses observer. 
 77        * 
 78        * Fires a changeObserverPause event 
 79        */
 80       pause : function() {
 81         if (this._editor.fire("changeObserverPause")) {
 82           this._paused = true;
 83         }
 84       },
 85       /**
 86        * Resumes observer
 87        * 
 88        * Fires a changeObserverResume event
 89        */
 90       resume : function() {
 91         if (this._editor.fire("changeObserverResume")) {
 92           this._paused = false;
 93           this.checkChange();
 94         }
 95       },
 96       _getEditorContent : function() {
 97         // TODO: Why editor.getSnapshot() keeps returning 'true'???
 98         // TODO: Data should be configurable
 99         return this.getEditor().getData();
100       }
101     }
102   });
103 
104   /**
105    * Change observer that uses MutationObserver for change detection.
106    * 
107    * @class
108    * @extends ChangeObserver
109    */
110   MutationChangeObserver = CKEDITOR.tools.createClass({
111     base : ChangeObserver,
112     /**
113      * Creates a MutationChangeObserver class instance.
114      *
115      * @constructor
116      * @param {CKEDITOR.editor} editor object observer observes.
117      */
118     $ : function(editor) {
119       this.base(editor);
120 
121       var MutationObserver = window.MutationObserver || window.WebKitMutationObserver || window.MozMutationObserver;
122 
123       var _this = this;
124       this._observer = new MutationObserver(function(mutations) {
125         CKEDITOR.tools.setTimeout(function() {
126           this.checkChange();
127         }, 0, _this);
128       });
129     },
130     proto : /** @lends MutationChangeObserver.prototype */ {
131       /**
132        * Starts observer
133        */
134       start : function() {
135         var editorDocument = this._editor.document.$;
136 
137         this._observer.observe(editorDocument, {
138           attributes : true,
139           childList : true,
140           characterData : true,
141           subtree : true
142         });
143       },
144       /**
145        * Stops observer
146        */
147       disconnect : function() {
148         this._observer.disconnect();
149         this._observer = undefined;
150       }
151     }
152   });
153 
154   /**
155    * Change observer that uses DOMSubtreeModified event for change detection.
156    * 
157    * @class
158    * @extends ChangeObserver
159    */
160   DOMSubtreeModifiedChangeObserver = CKEDITOR.tools.createClass({
161     base : ChangeObserver,
162     /**
163      * Creates a DOMSubtreeModifiedChangeObserver class instance.
164      *
165      * @constructor
166      * @param {CKEDITOR.editor} editor object observer observes.
167      */
168     $ : function(editor) {
169       this.base(editor);
170 
171       var _this = this;
172       this._domSubtreeModifiedListener = function(event) {
173         _this._onDOMSubtreeModified(event);
174       };
175     },
176     proto : /** @lends DOMSubtreeModifiedChangeObserver.prototype */ {
177       /**
178        * Starts observer
179        */
180       start : function() {
181         this._editor.document.on("DOMSubtreeModified", this._domSubtreeModifiedListener);
182       },
183       /**
184        * Stops observer
185        */
186       disconnect : function() {
187         this._editor.document.removeListener("DOMSubtreeModified", this._domSubtreeModifiedListener);
188       },
189       _onDOMSubtreeModified : function(event) {
190         CKEDITOR.tools.setTimeout(function() {
191           this.checkChange();
192         }, 0, this);
193       }
194     }
195   });
196   
197   /**
198    * Change observer that uses polling for change detection.
199    * 
200    * @class
201    * @extends ChangeObserver
202    */
203   PollingChangeObserver = CKEDITOR.tools.createClass({
204     base : ChangeObserver,
205     /**
206      * Creates a PollingChangeObserver class instance.
207      *
208      * @constructor
209      * @param {CKEDITOR.editor} editor object observer observes.
210      */
211     $ : function(editor) {
212       this.base(editor);
213       this._timer = null;
214     },
215     proto : /** @lends PollingChangeObserver.prototype */ {
216       /**
217        * Starts observer
218        */      
219       start : function() {
220         this._poll();
221       },
222       /**
223        * Stops observer
224        */
225       disconnect : function() {
226         if (this._timer) {
227           clearTimeout(this._timer);
228         }
229 
230         this._timer = null;
231       },
232       _poll : function(event) {
233         this.checkChange();
234         this._timer = CKEDITOR.tools.setTimeout(this._poll, 333, this);
235       }
236     }
237   });
238 
239   CKEDITOR.plugins.add('change', {
240     requires : [],
241     init : function(editor) {
242       CKEDITOR.tools.extend(CKEDITOR.editor.prototype, {
243         /**
244          * Returns change observer
245          * 
246          * @returns {ChangeObserver} change observer
247          */
248         getChangeObserver : function() {
249           return editor._changeObserver;
250         }
251       });
252 
253       var changeObserver = null;
254       if ((typeof (window.MutationObserver || window.WebKitMutationObserver || window.MozMutationObserver) == 'function')) {
255         // Modern browsers support mutation observer
256         changeObserver = new MutationChangeObserver(editor);
257       }
258 
259       if (changeObserver == null) {
260         if (CKEDITOR.env.webkit || (CKEDITOR.env.ie && CKEDITOR.env.version >= 9)) {
261           // WebKit and IE 9 support DOMSubtreeModified
262           changeObserver = new DOMSubtreeModifiedChangeObserver(editor);
263         } else {
264           // Otherwise we fallback to polling
265           changeObserver = new PollingChangeObserver(editor);
266         }
267       }
268 
269       editor._changeObserver = changeObserver;
270 
271       editor.on('instanceReady', function() {
272         this.getChangeObserver().start();
273       });
274 
275       editor.on('beforePaste', function(event) {
276         this.getChangeObserver().pause();
277       });
278 
279       editor.on('paste', function(event) {
280         this.getChangeObserver().resume();
281       }, editor, null, 99999);
282     }
283   });
284 
285 }).call(this);
